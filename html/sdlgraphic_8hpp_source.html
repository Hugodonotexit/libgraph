108void SDLG::eventLoop() {
109 // Handle events on queue
110 while (SDL_PollEvent(&e) != 0) {
111 // User requests quit
112 switch (e.type)
113 {
114 case SDL_QUIT:
115 isOpen = false;
116 break;
117 case SDL_MOUSEWHEEL:
118 double scale = 1 + e.wheel.y;
119 updateCurve(scale);
120 break;
121 }
122 }
123}

131void SDLG::updateLine(double scale) {
132 for (int i = 0; i < (int)lines.size(); i++) {
133 lines[i].first.first = lines[i].first.first * scale;
134 lines[i].first.second = lines[i].first.second * scale;
135 }

138void SDLG::updateCurve(double scale) {
139 for (int i = 0; i < (int)curves.size(); i++) {
140 for (int j = 0; j < (int)curves[i].size(); j++)
141 {
142 curves[i][j].first.first = curves[i][j].first.first * scale;
143 curves[i][j].first.second = curves[i][j].first.second * scale;
144 }
145 }
146}

148void SDLG::drawLines() {
149 for (int i = 0; i < (int)lines.size(); i++) {
150 SDL_SetRenderDrawColor(renderer, lines[i].second.r, lines[i].second.g,
151 lines[i].second.b, lines[i].second.t);
152
153 SDL_RenderDrawLine(renderer, lines[i].first.first.x, lines[i].first.first.y,
154 lines[i].first.second.x, lines[i].first.second.y);
155 }
156}
157void SDLG::drawCurve() {
158 for (int i = 0; i < (int)curves.size(); i++) {
159 for (int j = 0; j < (int)curves[i].size(); j++) {
160 SDL_SetRenderDrawColor(renderer, curves[i][j].second.r, curves[i][j].second.g,
161 curves[i][j].second.b, curves[i][j].second.t);
162
163 SDL_RenderDrawLine(renderer, curves[i][j].first.first.x, curves[i][j].first.first.y,
164 curves[i][j].first.second.x, curves[i][j].first.second.y);
165 }
166 }

239void SDLG::setCurve(int funcIndex, Colour colour) {
240 std::vector<std::pair<LineSeg, Colour>> arc;
241 for (double i = 0; i < 1000; i = i + 0.5) {
242 LineSeg lineSeg =
243 std::make_pair(Vectorlf(i, funcs[funcIndex].get_y(i)),
244 Vectorlf(i + 1, funcs[funcIndex].get_y(i + 1)));
245 arc.push_back(std::make_pair(lineSeg,colour));
246 }
247 curves.push_back(arc);
248}
